-- MÃ HÓA TOÀN BỘ CHUỖI VÀ BIẾN HỆ THỐNG
local _0x1 = {
    ["K"] = {67, 104, 105, 105, 72, 97, 117, 100, 122}, -- ChiiHaudz
    ["S"] = 175,
    ["H"] = "https://sirius.menu/rayfield",
}

-- Giải mã Key nội bộ (Không hiển thị ra ngoài)
local function _GET_KEY(t)
    local k = ""
    for i = 1, #t do k = k .. string.char(t[i]) end
    return k
end

local _S = _0x1["S"]
local _K = _GET_KEY(_0x1["K"])
local _A = false
local _G_S = game.GetService
local _TS, _RS, _HS, _TPS = _G_S(game, "TweenService"), _G_S(game, "RunService"), _G_S(game, "HttpService"), _G_S(game, "TeleportService")
local _LP = _G_S(game, "Players").LocalPlayer

local Rayfield = loadstring(game:HttpGet(_0x1["H"]))()
_G.AutoFarmChest = false
local _CT = nil

-- HÀM FIX GIẬT: Đồng bộ hóa vật lý tuyệt đối
local function _STB()
    local c = _LP.Character
    local r = c and c:FindFirstChild("HumanoidRootPart")
    if r then
        local bv = r:FindFirstChild("_CH_BV") or Instance.new("BodyVelocity", r)
        bv.Name = "_CH_BV"
        bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
        bv.Velocity = Vector3.new(0, 0, 0)
        
        local bg = r:FindFirstChild("_CH_BG") or Instance.new("BodyGyro", r)
        bg.Name = "_CH_BG"
        bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
        bg.P = 20000 -- Tăng độ cứng để không bị lắc camera
        bg.CFrame = r.CFrame
    end
end

-- QUÉT RƯƠNG THÔNG MINH + TỰ ĐỔI MAP (MAP HOP)
local function _GET_C()
    local t, d = nil, math.huge
    local r = _LP.Character and _LP.Character:FindFirstChild("HumanoidRootPart")
    if not r then return nil end

    -- Ưu tiên quét các vật thể chứa "Chest" trong tên ở Workspace
    local descendants = workspace:GetDescendants()
    for i = 1, #descendants do
        local v = descendants[i]
        if v:IsA("BasePart") and v.Name:lower():find("chest") then
            -- Kiểm tra rương chưa bị mở (TouchInterest hoặc Transparency)
            if v:FindFirstChild("TouchInterest") or v.Transparency < 1 then
                local dist = (v.Position - r.Position).Magnitude
                if dist < d then
                    d = dist
                    t = v
                end
            end
        end
    end
    return t
end

-- SERVER HOP THÔNG MINH (KHI HẾT RƯƠNG)
local function _SERVER_HOP()
    Rayfield:Notify({Title = "THÔNG BÁO", Content = "Hết rương! Đang đổi Server mới...", Duration = 3})
    task.wait(1)
    pcall(function()
        local url = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
        local servers = _HS:JSONDecode(game:HttpGet(url))
        for _, s in pairs(servers.data) do
            if s.playing < s.maxPlayers and s.id ~= game.JobId then
                _TPS:TeleportToPlaceInstance(game.PlaceId, s.id)
                break
            end
        end
    end)
end

-- LOGIC FARM LIÊN TỤC
function _START_F()
    task.spawn(function()
        while _G.AutoFarmChest do
            task.wait()
            pcall(function()
                local r = _LP.Character and _LP.Character:FindFirstChild("HumanoidRootPart")
                if r then
                    local target = _GET_C()
                    if target then
                        _STB()
                        local dist = (r.Position - target.Position).Magnitude
                        -- Tween mượt không gia tốc (Linear)
                        if _CT then _CT:Cancel() end
                        _CT = _TS:Create(r, TweenInfo.new(dist / _S, Enum.EasingStyle.Linear), {CFrame = target.CFrame * CFrame.new(0, 4, 0)})
                        _CT:Play()
                        
                        -- Chờ nhặt xong rương (Timeout 8s tránh kẹt)
                        local start = tick()
                        repeat task.wait() 
                        until not target.Parent or not _G.AutoFarmChest or (tick() - start > 8)
                    else
                        -- Hết rương -> Tắt Farm và Hop
                        if _CT then _CT:Cancel() end
                        _G.AutoFarmChest = false
                        _SERVER_HOP()
                    end
                end
            end)
        end
    end)
end

-- NOCLIP + AIR STATE (CHỐNG GIẬT KHI VA CHẠM)
_RS.Stepped:Connect(function()
    if _G.AutoFarmChest then
        pcall(function()
            local char = _LP.Character
            if char then
                for _, v in pairs(char:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end
                if char:FindFirstChildOfClass("Humanoid") then
                    char:FindFirstChildOfClass("Humanoid"):ChangeState(11)
                end
            end
        end)
    end
end)

})

   end,
})
