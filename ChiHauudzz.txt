-- [[ CHII HAUU HUB - VERSION 8.0 GOD MODE ]]
local _DATA = {
    ["K"] = {67, 104, 105, 105, 72, 97, 117, 100, 122}, -- ChiiHaudz
    ["S"] = 285, 
    ["U"] = "https://sirius.menu/rayfield",
}

local function _DE(t)
    local k = ""
    for i = 1, #t do k = k .. string.char(t[i]) end
    return k
end

local _S, _K = _DATA["S"], _DE(_DATA["K"])
local _G_S = game.GetService
local _TS, _RS, _HS, _TPS = _G_S(game, "TweenService"), _G_S(game, "RunService"), _G_S(game, "HttpService"), _G_S(game, "TeleportService")
local _LP = _G_S(game, "Players").LocalPlayer
local _A = false

local Rayfield = loadstring(game:HttpGet(_DATA["U"]))()
_G.AutoFarmChest = false
local _CT = nil

-- HÀM DỌN DẸP HOÀN TOÀN (FIX LỖI KHÔNG DI CHUYỂN ĐƯỢC)
local function _CLEAN_UP()
    local c = _LP.Character
    local r = c and c:FindFirstChild("HumanoidRootPart")
    if r then
        if r:FindFirstChild("_INFINITY_V") then r._INFINITY_V:Destroy() end
        if r:FindFirstChild("_INFINITY_G") then r._INFINITY_G:Destroy() end
    end
    if c and c:FindFirstChild("Humanoid") then
        c.Humanoid.PlatformStand = false
        c.Humanoid:ChangeState(11) -- Reset state
        task.wait(0.1)
        c.Humanoid:ChangeState(8) -- Đưa về trạng thái Running
    end
    if _CT then _CT:Cancel() end
end

-- KIỂM TRA VẬT PHẨM QUÝ (KEY / CHÉN THÁNH)
local function _CHECK_RARE()
    local inv = _LP.Backpack:GetChildren()
    local char = _LP.Character:GetChildren()
    local all = {}
    for _, v in pairs(inv) do table.insert(all, v.Name:lower()) end
    for _, v in pairs(char) do table.insert(all, v.Name:lower()) end
    
    for _, name in pairs(all) do
        if name:find("key") or name:find("cup") or name:find("chalice") or name:find("holy") then
            return true
        end
    end
    return false
end

-- KHÓA VẬT LÝ KHI BAY
local function _LOCK(r)
    if not r:FindFirstChild("_INFINITY_V") then
        local bv = Instance.new("BodyVelocity", r)
        bv.Name = "_INFINITY_V"
        bv.MaxForce = Vector3.new(1e10, 1e10, 1e10)
        bv.Velocity = Vector3.new(0, 0, 0)
        
        local bg = Instance.new("BodyGyro", r)
        bg.Name = "_INFINITY_G"
        bg.MaxTorque = Vector3.new(1e10, 1e10, 1e10)
        bg.P = 100000
        bg.CFrame = r.CFrame
    end
    if _LP.Character:FindFirstChild("Humanoid") then
        _LP.Character.Humanoid.PlatformStand = true
    end
end

-- TÌM RƯƠNG SIÊU TỐC
local function _SCAN()
    local t, d = nil, math.huge
    local r = _LP.Character and _LP.Character:FindFirstChild("HumanoidRootPart")
    if not r then return nil end

    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and (v.Name:find("Chest") or v.Name:find("Box")) then
            if v:FindFirstChild("TouchInterest") or v.Transparency < 1 then
                local dist = (v.Position - r.Position).Magnitude
                if dist < d then d = dist; t = v end
            end
        end
    end
    return t
end

-- SERVER HOP TỨC THÌ
local function _QUICK_HOP()
    pcall(function()
        local r = game:HttpGet("https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100")
        local s = _HS:JSONDecode(r).data
        for _, v in pairs(s) do
            if v.playing < v.maxPlayers and v.id ~= game.JobId then
                _TPS:TeleportToPlaceInstance(game.PlaceId, v.id)
                break
            end
        end
    end)
end

-- LOGIC FARM KHÔNG DỪNG VÀ TỰ DỪNG KHI CÓ ITEM QUÝ
function _INFINITY_FARM()
    task.spawn(function()
        while _G.AutoFarmChest do
            _RS.Heartbeat:Wait()
            
            -- KIỂM TRA ITEM QUÝ TRƯỚC MỖI LẦN NHẶT
            if _CHECK_RARE() then
                _G.AutoFarmChest = false
                _CLEAN_UP()
                Rayfield:Notify({Title = "⚠️ CẢNH BÁO", Content = "Đã nhặt được vật phẩm quý! Tự động dừng để bảo vệ.", Duration = 10})
                break
            end

            local r = _LP.Character and _LP.Character:FindFirstChild("HumanoidRootPart")
            if r then
                local chest = _SCAN()
                if chest then
                    _LOCK(r)
                    local dist = (r.Position - chest.Position).Magnitude
                    if _CT then _CT:Cancel() end
                    _CT = _TS:Create(r, TweenInfo.new(dist / _S, Enum.EasingStyle.Linear), {CFrame = chest.CFrame})
                    _CT:Play()
                    
                    while chest.Parent and _G.AutoFarmChest and (r.Position - chest.Position).Magnitude > 5 do
                        _RS.Heartbeat:Wait()
                        if _CHECK_RARE() then break end -- Kiểm tra tức thì khi đang bay
                    end
                else
                    _CLEAN_UP()
                    _QUICK_HOP()
                    break
                end
            else
                task.wait(1)
            end
        end
        _CLEAN_UP()
    end)
end

-- NOCLIP SIÊU MƯỢT
_RS.Stepped:Connect(function()
    if _G.AutoFarmChest and _LP.Character then
        for _, v in pairs(_LP.Character:GetDescendants()) do
            if v:IsA("BasePart") then v.CanCollide = false end
        end
        local h = _LP.Character:FindFirstChild("Humanoid")
        if h then h:ChangeState(11) end
    end
end)

local Window = Rayfield:CreateWindow({
   Name = "🛡️ Chii Hauu HUB | V8 GOD MODE",
   LoadingTitle = "Ultimate Protection System",
   LoadingSubtitle = "By Chii Hauu",
   KeySystem = false
})

local Tab = Window:CreateTab("💰 Farm")

Tab:CreateInput({
   Name = "Mã xác nhận",
   PlaceholderText = "Nhập mã...",
   Callback = function(t)
      if t == _K then 
         _A = true 
         Rayfield:Notify({Title = "OK", Content = "Xác minh thành công!", Duration = 3})
      end
   end,
})

Tab:CreateToggle({
   Name = "Auto Collect Infinity (Stop at Rare)",
   CurrentValue = false,
   Callback = function(v)
      if _A then
         _G.AutoFarmChest = v
         if v then _INFINITY_FARM() else _CLEAN_UP() end
      else
         Rayfield:Notify({Title = "LỖI", Content = "Vui lòng nhập mã!", Duration = 3})
      end
   end,
})

Tab:CreateButton({
   Name = "🛑 Stop & Clean",
   Callback = function()
      _G.AutoFarmChest = false
      _CLEAN_UP()
      Rayfield:Destroy()
   end,
})
